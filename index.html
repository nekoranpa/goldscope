function buildEASignal(r){ const rr = r.reboundRisk || {}; return { enabled: true, symbol: 'XAUUSD', action: r.signal?.mainDir === 'up' ? 'BUY' : r.signal?.mainDir === 'dn' ? 'SELL' : 'NONE', price: parseFloat(r.price.toFixed(2)), score: r.utbotScore || r.entryQuality?.passCount || 0, confidence: r.signal?.mainDir === 'up' ? r.signal?.upPct : r.signal?.dnPct, live: r.spotMeta?.live === true, source: r.spotMeta?.source || 'unknown', reboundRisk: rr.score || 0, reboundDecision: rr.decision || 'normal', autoTradeAllowed: rr.autoTradeAllowed !== false, lotMultiplier: rr.lotMultiplier ?? 1.0, tpMultiplier: rr.tpMultiplier ?? 1.0, nearestWall: rr.nearestWall?.label || null, wallPrice: rr.nearestWall?.wallPrice || null, wallDistance: rr.nearestWall?.distance || null, expiresAt: Date.now() + 120000 }; }
